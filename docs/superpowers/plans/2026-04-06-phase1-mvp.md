# BuilderOS Phase 1 MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a white-label post-closing handoff platform where builders create templated home packages, invite buyers via magic link, and buyers complete warranty registrations with deadline-driven reminders.

**Architecture:** Next.js App Router with Supabase (Postgres + Auth + Storage). Server actions for mutations, server components for data fetching. Inngest for scheduled reminder jobs. Multi-tenant with RLS. White-label via CSS custom properties loaded from builder branding config.

**Tech Stack:** Next.js 15 (App Router), Supabase (Postgres, Auth, Storage), Inngest, Resend, Tailwind CSS, shadcn/ui, TypeScript, Vitest

**Spec reference:** `docs/superpowers/specs/2026-04-06-post-close-handoff-platform-design.md`

```sql
-- Profiles (extends Supabase Auth)
CREATE TABLE profiles (
  id              uuid PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  email           text UNIQUE NOT NULL,
  full_name       text,
  avatar_url      text,
  created_at      timestamptz DEFAULT now(),
  updated_at      timestamptz DEFAULT now()
);

-- Auto-create profile on auth signup
CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS trigger AS $$
BEGIN
  INSERT INTO public.profiles (id, email, full_name)
  VALUES (NEW.id, NEW.email, NEW.raw_user_meta_data->>'full_name');
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION handle_new_user();

-- Builders (tenants)

  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name            text NOT NULL,
  slug            text UNIQUE NOT NULL,

  accent_color    text DEFAULT '#2563eb',

-- Profile: users can read/update their own profile
CREATE POLICY "Users can read own profile"
  ON profiles FOR SELECT
  USING (auth.uid() = id);

CREATE POLICY "Users can update own profile"
  ON profiles FOR UPDATE
  USING (auth.uid() = id);

-- Builders: members can read their builder
CREATE POLICY "Members can read their builder"
  ON builders FOR SELECT
  USING (
    EXISTS (
      SELECT 1 FROM memberships
      WHERE memberships.builder_id = builders.id
      AND memberships.user_id = auth.uid()
    )
  );

-- Builders: owners can update their builder
CREATE POLICY "Owners can update their builder"
  ON builders FOR UPDATE
  USING (
    EXISTS (
      SELECT 1 FROM memberships
      WHERE memberships.builder_id = builders.id
      AND memberships.user_id = auth.uid()
      AND memberships.role = 'owner'
    )
  );

-- Memberships: users can read their own memberships
CREATE POLICY "Users can read own memberships"
  ON memberships FOR SELECT
  USING (user_id = auth.uid());

-- Memberships: owners can read all memberships for their builder
CREATE POLICY "Owners can read builder memberships"
  ON memberships FOR SELECT
  USING (
    EXISTS (
      SELECT 1 FROM memberships AS m
      WHERE m.builder_id = memberships.builder_id
      AND m.user_id = auth.uid()
      AND m.role = 'owner'
    )
  );

-- Projects: builder members can read projects
CREATE POLICY "Members can read projects"
  ON projects FOR SELECT
  USING (
    EXISTS (
      SELECT 1 FROM memberships
      WHERE memberships.builder_id = projects.builder_id
      AND memberships.user_id = auth.uid()
      AND memberships.role IN ('owner', 'staff')

-- Projects: owners can manage projects
```

```typescript
import { createServerClient } from "@supabase/ssr";
import { NextResponse, type NextRequest } from "next/server";

export async function middleware(request: NextRequest) {
  let supabaseResponse = NextResponse.next({ request });

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return request.cookies.getAll();
        },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value, options }) =>
            request.cookies.set(name, value)
          );
          supabaseResponse = NextResponse.next({ request });
          cookiesToSet.forEach(({ name, value, options }) =>
            supabaseResponse.cookies.set(name, value, options)
          );
        },
      },
    }
  );

  const {
    data: { user },
  } = await supabase.auth.getUser();

  // Protected builder routes
  if (request.nextUrl.pathname.startsWith("/dashboard") ||
      request.nextUrl.pathname.startsWith("/settings") ||
      request.nextUrl.pathname.startsWith("/projects") ||
      request.nextUrl.pathname.startsWith("/templates") ||
      request.nextUrl.pathname.startsWith("/homes")) {
    if (!user) {
      const url = request.nextUrl.clone();
      url.pathname = "/login";
      return NextResponse.redirect(url);
    }
  }

  // Protected buyer routes
  if (request.nextUrl.pathname.startsWith("/home/")) {
    if (!user) {
      const url = request.nextUrl.clone();
      url.pathname = "/login";
      url.searchParams.set("redirect", request.nextUrl.pathname);
      return NextResponse.redirect(url);
    }
  }

  return supabaseResponse;
}

export const config = {
  matcher: [
    "/((?!_next/static|_next/image|favicon.ico|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)",
  ],
};
```

```typescript
function generateSlug(name: string): string {
  return name
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, "-")
    .replace(/^-|-$/g, "")
    .slice(0, 50) + "-" + Math.random().toString(36).slice(2, 6);
}

export async function signUp(formData: FormData) {
  const supabase = await createClient();

  const email = formData.get("email") as string;
  const password = formData.get("password") as string;
  const fullName = formData.get("fullName") as string;
  const companyName = formData.get("companyName") as string;

  // 1. Create auth user
  const { data: authData, error: authError } = await supabase.auth.signUp({
    email,
    password,
    options: {
      data: { full_name: fullName },
    },
  });

  if (authError || !authData.user) {
    return { error: authError?.message || "Signup failed" };
  }

  // 2. Create builder tenant
  const { data: builder, error: builderError } = await supabase
    .from("builders")
    .insert({
      name: companyName,
      slug: generateSlug(companyName),
      contact_email: email,
    })
    .select()
    .single();

  if (builderError) {
    return { error: "Failed to create builder account" };
  }

  // 3. Create owner membership
  const { error: membershipError } = await supabase
    .from("memberships")
    .insert({
      user_id: authData.user.id,
      builder_id: builder.id,
      role: "owner",
    });

  if (membershipError) {
    return { error: "Failed to create membership" };
  }

  redirect("/dashboard");
}

export async function signIn(formData: FormData) {
  const supabase = await createClient();

  const email = formData.get("email") as string;
  const password = formData.get("password") as string;

  const { error } = await supabase.auth.signInWithPassword({
    email,
    password,
  });

  if (error) {
    return { error: error.message };
  }

  redirect("/dashboard");
}

export async function signOut() {
  const supabase = await createClient();
  await supabase.auth.signOut();
  redirect("/login");
}
```

```typescript
import { createClient } from "@/lib/supabase/server";
```

```typescript
export async function updateBuilderSettings(formData: FormData) {
  const supabase = await createClient();
  const context = await getCurrentBuilder();

  if (!context || context.role !== "owner") {
    return { error: "Unauthorized" };
  }

  const { error } = await supabase
    .from("builders")

      updated_at: new Date().toISOString(),
    })
    .eq("id", context.builder.id);

  if (error) {
    return { error: "Failed to update settings" };
  }

  revalidatePath("/settings");
  return { success: true };
}
```

```typescript
export async function getTemplates() {
  const supabase = await createClient();

  if (!context) return [];

    .from("templates")
    .select("*, template_items(count)")
    .eq("builder_id", context.builder.id)
    .order("created_at", { ascending: false });

  return data || [];

  const supabase = await createClient();
  const context = await getCurrentBuilder();
  if (!context) return null;

  const { data } = await supabase
    .from("templates")
    .select("*, template_items(*)")
    .eq("id", templateId)
    .eq("builder_id", context.builder.id)
    .single();

  if (data?.template_items) {
    data.template_items.sort((a: any, b: any) => a.sort_order - b.sort_order);
  }

  return data;
}
```

```typescript
  const context = await getCurrentBuilder();
  if (!context || context.role !== "owner") {
    return { error: "Unauthorized" };
```

```typescript
  const item: Record<string, any> = {

    category: formData.get("category") as string,
    title: formData.get("title") as string,

    item.manufacturer = formData.get("manufacturer") as string || null;
    item.registration_url = formData.get("registrationUrl") as string || null;
    item.registration_deadline_offset = formData.get("registrationDeadlineOffset")
      ? Number(formData.get("registrationDeadlineOffset"))
```

```tsx
export default async function TemplateDetailPage({
  params,
}: {
  params: Promise<{ templateId: string }>;
}) {
  const { templateId } = await params;
  const template = await getTemplate(templateId);

  if (!template) {
    notFound();
  }

  const updateWithId = updateTemplate.bind(null, templateId);
  const deleteWithId = deleteTemplate.bind(null, templateId);
  const createItemWithId = createTemplateItem.bind(null, templateId);

  // Group items by category
  const categories = (template.template_items || []).reduce((acc: Record<string, any[]>, item: any) => {
    const cat = item.category || "Uncategorized";
    if (!acc[cat]) acc[cat] = [];
    acc[cat].push(item);
    return acc;
  }, {});

  return (
    <div className="space-y-6">
      <div className="flex items-center justify-between">
        <h1 className="text-2xl font-bold">{template.name}</h1>
        <form action={deleteWithId}>
          <Button variant="destructive" size="sm" type="submit">Delete Template</Button>
        </form>
      </div>

        <CardHeader>
          <CardTitle>Template Settings</CardTitle>

            <Button size="sm">Add Item</Button>
          </DialogTrigger>
          <DialogContent className="max-w-2xl max-h-[80vh] overflow-y-auto">
            <DialogHeader>
              <DialogTitle>Add Template Item</DialogTitle>
            </DialogHeader>
            <ItemForm action={createItemWithId} />
          </DialogContent>
        </Dialog>
      </div>

      {Object.keys(categories).length === 0 ? (
        <p className="text-muted-foreground">No items yet. Add items to build your handoff template.</p>

                  );
                })}
              </CardContent>
            </Card>
          ))}
        </div>
      )}
    </div>
  );
}
```

```typescript
type ItemForCompletion = {
  is_critical: boolean;
  status: string;
};

const RESOLVED_STATUSES = new Set(["complete", "skipped", "not_applicable"]);

export function calculateCompletion(items: ItemForCompletion[]): number {
  const criticalItems = items.filter((item) => item.is_critical);

  if (criticalItems.length === 0) return 100;

  const resolved = criticalItems.filter((item) =>
    RESOLVED_STATUSES.has(item.status)
  ).length;

  return Math.floor((resolved / criticalItems.length) * 100);
}
```

```typescript
export async function getHomes(filters?: { projectId?: string; status?: string }) {
  const supabase = await createClient();
  const context = await getCurrentBuilder();
  if (!context) return [];

  let query = supabase
    .from("homes")
    .select("*, projects(name), invitations(status)")
    .eq("builder_id", context.builder.id)
    .order("created_at", { ascending: false });

  if (filters?.projectId) {
    query = query.eq("project_id", filters.projectId);
  }

  if (filters?.status) {
    query = query.eq("handoff_status", filters.status);
  }

  const { data } = await query;
  return data || [];
}

export async function getHome(homeId: string) {
  const supabase = await createClient();
  const context = await getCurrentBuilder();
  if (!context) return null;

  const { data } = await supabase
    .from("homes")
    .select("*, projects(name), home_items(*), invitations(*), home_assignments(*, profiles(full_name, email))")
    .eq("id", homeId)
    .eq("builder_id", context.builder.id)
    .single();

  return data;
}
```

```typescript
"use server";

import { createClient } from "@/lib/supabase/server";
import { getCurrentBuilder } from "@/lib/queries/builders";
import { revalidatePath } from "next/cache";
import { redirect } from "next/navigation";

export async function createHome(formData: FormData) {
  const supabase = await createClient();
  const context = await getCurrentBuilder();
  if (!context || context.role !== "owner") {
    return { error: "Unauthorized" };
  }

  const templateId = formData.get("templateId") as string;
  const closeDate = formData.get("closeDate") as string || null;
  const projectId = formData.get("projectId") as string || null;

  // 1. Create the home
  const { data: home, error: homeError } = await supabase
    .from("homes")
    .insert({
      builder_id: context.builder.id,
      template_id: templateId || null,
      project_id: projectId || null,
      address: formData.get("address") as string,
      lot_number: formData.get("lotNumber") as string || null,
      close_date: closeDate,
      handoff_status: "draft",
    })
    .select()
    .single();

  if (homeError || !home) {
    return { error: "Failed to create home" };
  }

  // 2. Clone template items if a template was selected
  if (templateId) {
    const { data: templateItems } = await supabase
      .from("template_items")
      .select("*")
      .eq("template_id", templateId);

    if (templateItems && templateItems.length > 0) {
      const closeDateObj = closeDate ? new Date(closeDate) : null;

      const homeItems = templateItems.map((ti) => {
        const item: Record<string, any> = {
          home_id: home.id,
          builder_id: context.builder.id,
          type: ti.type,
          category: ti.category,
          title: ti.title,
          description: ti.description,
          sort_order: ti.sort_order,
          status: "pending",
          source: "template",
          template_item_id: ti.id,
          is_critical: ti.is_critical,
          metadata: ti.metadata,
        };

        // Compute absolute due date from offset
        if (ti.due_date_offset && closeDateObj) {
          const dueDate = new Date(closeDateObj);
          dueDate.setDate(dueDate.getDate() + ti.due_date_offset);
          item.due_date = dueDate.toISOString().split("T")[0];
        }

        // Warranty fields
        if (ti.type === "warranty") {
          item.manufacturer = ti.manufacturer;
          item.registration_url = ti.registration_url;
          item.responsible_party = ti.responsible_party;
          item.registration_status = "not_started";
          if (ti.registration_deadline_offset && closeDateObj) {
            const deadline = new Date(closeDateObj);
            deadline.setDate(deadline.getDate() + ti.registration_deadline_offset);
            item.registration_deadline = deadline.toISOString().split("T")[0];
          }
        }

        // Utility fields
        if (ti.type === "utility") {
          item.utility_type = ti.utility_type;
        }

        return item;
      });

      await supabase.from("home_items").insert(homeItems);

      // Clone template-level files
      const { data: templateFiles } = await supabase
        .from("files")
        .select("*")
        .eq("template_id", templateId);

      if (templateFiles && templateFiles.length > 0) {
        const homeFiles = templateFiles.map((tf) => ({
          builder_id: context.builder.id,
          home_id: home.id,
          home_item_id: null, // Will need to map template_item_id -> home_item_id if needed
          uploaded_by: tf.uploaded_by,
          storage_path: tf.storage_path, // Same file, new metadata row
          filename: tf.filename,
          mime_type: tf.mime_type,
          size_bytes: tf.size_bytes,
        }));

        await supabase.from("files").insert(homeFiles);
      }
    }
  }

  // 3. Log activity
  await supabase.from("activity_log").insert({
    builder_id: context.builder.id,
    home_id: home.id,
    actor_type: "user",
    actor_id: context.userId,
    action: "home_created",
    metadata: { template_id: templateId },
  });

  revalidatePath("/homes");
  redirect(`/homes/${home.id}`);
}

export async function updateHomeStatus(homeId: string, status: string) {
  const supabase = await createClient();
  const context = await getCurrentBuilder();
  if (!context || context.role !== "owner") {
    return { error: "Unauthorized" };
  }

  const { error } = await supabase
    .from("homes")
    .update({
      handoff_status: status,
      updated_at: new Date().toISOString(),
    })
    .eq("id", homeId)
    .eq("builder_id", context.builder.id);

  if (error) {
    return { error: "Failed to update status" };
  }

  await supabase.from("activity_log").insert({
    builder_id: context.builder.id,
    home_id: homeId,
    actor_type: "user",
    actor_id: context.userId,
    action: "status_changed",
    metadata: { new_status: status },
  });

  revalidatePath(`/homes/${homeId}`);
  return { success: true };
}
```

```typescript
export async function updateHomeItemStatus(homeId: string, itemId: string, status: string) {
  const supabase = await createClient();
  const context = await getCurrentBuilder();

  const updates: Record<string, any> = {
    status,
    updated_at: new Date().toISOString(),

  // If warranty item marked as registered
  if (status === "complete") {
    const { data: item } = await supabase

    .from("home_items")
    .update(updates)
    .eq("id", itemId)
    .eq("home_id", homeId);

    return { error: "Failed to update item" };
  }

  // Recalculate completion
  const { data: allItems } = await supabase
    .from("home_items")
    .select("is_critical, status")
    .eq("home_id", homeId);

      completion_pct: pct,
      updated_at: new Date().toISOString(),

    // Auto-complete home if 100%

      updates.handoff_status = "completed";

    await supabase.from("homes").update(updates).eq("id", homeId);
  }

  // Log activity
  const actorId = context?.userId;

    builder_id: context?.builder.id,
    home_id: homeId,
    home_item_id: itemId,
```

```typescript
"use server";

import { createClient } from "@/lib/supabase/server";
import { getCurrentBuilder } from "@/lib/queries/builders";
import { resend } from "@/lib/email/client";
import { InvitationEmail } from "@/lib/email/templates/invitation";
import { revalidatePath } from "next/cache";

export async function sendInvitation(homeId: string, formData: FormData) {
  const supabase = await createClient();
  const context = await getCurrentBuilder();
  if (!context || context.role !== "owner") {
    return { error: "Unauthorized" };
  }

  const email = formData.get("email") as string;
  const role = (formData.get("role") as string) || "primary_buyer";

  // Get home details
  const { data: home } = await supabase
    .from("homes")
    .select("*")
    .eq("id", homeId)
    .eq("builder_id", context.builder.id)
    .single();

  if (!home) {
    return { error: "Home not found" };
  }

  // Create invitation
  const { data: invitation, error: invError } = await supabase
    .from("invitations")
    .insert({
      home_id: homeId,
      builder_id: context.builder.id,
      email,
      role,
      status: "pending",
    })
    .select()
    .single();

  if (invError || !invitation) {
    return { error: "Failed to create invitation" };
  }

  // Send email
  const acceptUrl = `${process.env.NEXT_PUBLIC_APP_URL}/accept-invite?token=${invitation.token}`;

  try {
    await resend.emails.send({
      from: `${context.builder.name} via BuilderOS <noreply@${process.env.RESEND_DOMAIN || "resend.dev"}>`,
      to: email,
      subject: `Your new home at ${home.address} is ready`,
      react: InvitationEmail({
        builderName: context.builder.name,
        homeAddress: home.address,
        acceptUrl,
        primaryColor: context.builder.primary_color || "#1a1a1a",
      }),
    });

    await supabase
      .from("invitations")
      .update({ status: "sent", sent_at: new Date().toISOString() })
      .eq("id", invitation.id);
  } catch {
    // Email failed but invitation exists — builder can resend
    return { error: "Invitation created but email failed to send. Try resending." };
  }

  // Update home status to invited if this is the first invitation
  if (home.handoff_status === "ready") {
    await supabase
      .from("homes")
      .update({ handoff_status: "invited", updated_at: new Date().toISOString() })
      .eq("id", homeId);
  }

  // Log activity
  await supabase.from("activity_log").insert({
    builder_id: context.builder.id,
    home_id: homeId,
    actor_type: "user",
    actor_id: context.userId,
    action: "invitation_sent",
    metadata: { email, role },
  });

  revalidatePath(`/homes/${homeId}`);
  return { success: true };
}

export async function resendInvitation(homeId: string, invitationId: string) {
  const supabase = await createClient();
```

Create `src/lib/utils/priority.ts`:

```typescript
type RankableItem = {
  id: string;
  due_date: string | null;
  is_critical: boolean;
  status: string;
};

const RESOLVED_STATUSES = new Set(["complete", "skipped", "not_applicable"]);

export function rankItems(items: RankableItem[], today: Date = new Date()): RankableItem[] {
  const active = items.filter((item) => !RESOLVED_STATUSES.has(item.status));

  return active.sort((a, b) => {
    const aDays = daysUntil(a.due_date, today);
    const bDays = daysUntil(b.due_date, today);

    // Items with due dates before items without
    if (aDays !== null && bDays === null) return -1;
    if (aDays === null && bDays !== null) return 1;

    // Both have due dates — sort by urgency (lowest days first)
    if (aDays !== null && bDays !== null) {
      if (aDays !== bDays) return aDays - bDays;
    }

    // Equal urgency — critical first
    if (a.is_critical !== b.is_critical) return a.is_critical ? -1 : 1;

    return 0;
  });
}

function daysUntil(dateStr: string | null, today: Date): number | null {
  if (!dateStr) return null;
  const target = new Date(dateStr);
  const diffMs = target.getTime() - today.getTime();
  return Math.floor(diffMs / (1000 * 60 * 60 * 24));
}
```

```typescript
import { createClient } from "@/lib/supabase/server";
import { calculateCompletion } from "@/lib/utils/completion";

export async function markItemComplete(homeId: string, itemId: string) {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { error: "Not authenticated" };

  // Verify access
  const { data: assignment } = await supabase
    .from("home_assignments")
    .select("*")
    .eq("home_id", homeId)
    .eq("user_id", user.id)
    .single();

  if (!assignment) return { error: "Access denied" };

  const { data: item } = await supabase
    .from("home_items")
    .select("type")
    .eq("id", itemId)
    .single();

  const updates: Record<string, any> = {
    status: "complete",
    updated_at: new Date().toISOString(),
  };

  if (item?.type === "warranty") {
    updates.registration_status = "registered";
  }

  await supabase
    .from("home_items")
    .update(updates)
    .eq("id", itemId)
    .eq("home_id", homeId);

  // Recalculate completion
  const { data: allItems } = await supabase
    .from("home_items")
    .select("is_critical, status")
    .eq("home_id", homeId);

  if (allItems) {
    const pct = calculateCompletion(allItems);
    const homeUpdates: Record<string, any> = {
      completion_pct: pct,
      updated_at: new Date().toISOString(),
    };

    // Check for first engagement
    const { data: home } = await supabase
      .from("homes")
      .select("handoff_status")
      .eq("id", homeId)
      .single();

    if (home?.handoff_status === "activated") {
      homeUpdates.handoff_status = "engaged";
    }
    if (pct === 100) {
      homeUpdates.handoff_status = "completed";
    }

    await supabase.from("homes").update(homeUpdates).eq("id", homeId);
  }

  // Log activity
  const { data: homeData } = await supabase
    .from("homes")
    .select("builder_id")
    .eq("id", homeId)
    .single();

  if (homeData) {
    await supabase.from("activity_log").insert({
      builder_id: homeData.builder_id,
      home_id: homeId,
      home_item_id: itemId,
      actor_type: "user",
      actor_id: user.id,
      action: "item_completed",
    });
  }

  revalidatePath(`/home/${homeId}`);
  revalidatePath(`/home/${homeId}/items/${itemId}`);
  return { success: true };
}
```

```typescript
export async function uploadFile(homeId: string, itemId: string | null, formData: FormData) {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { error: "Not authenticated" };

  const file = formData.get("file") as File;
  if (!file) return { error: "No file provided" };

  if (file.size > 25 * 1024 * 1024) {
    return { error: "File too large. Maximum 25MB." };
  }

  // Get builder_id from home
  const { data: home } = await supabase
    .from("homes")
    .select("builder_id")
    .eq("id", homeId)
    .single();

  if (!home) return { error: "Home not found" };

  const storagePath = `${home.builder_id}/${homeId}/${itemId || "general"}/${Date.now()}-${file.name}`;

  // Upload to Supabase Storage
  const { error: uploadError } = await supabase.storage
    .from("documents")
    .upload(storagePath, file);

  if (uploadError) {
    return { error: "Failed to upload file" };
  }

  // Create file record
  const { error: dbError } = await supabase
    .from("files")
    .insert({
      builder_id: home.builder_id,
      home_id: homeId,
      home_item_id: itemId,
      uploaded_by: user.id,
      storage_path: storagePath,
      filename: file.name,
      mime_type: file.type,
      size_bytes: file.size,
    });

  if (dbError) {
    return { error: "Failed to save file record" };
  }

  // Log activity
  await supabase.from("activity_log").insert({
    builder_id: home.builder_id,
    home_id: homeId,
    home_item_id: itemId,
    actor_type: "user",
    actor_id: user.id,
    action: "file_uploaded",
    metadata: { filename: file.name },
  });

  revalidatePath(`/homes/${homeId}`);
  revalidatePath(`/home/${homeId}`);
  revalidatePath(`/home/${homeId}/documents`);
  return { success: true };
}

export async function getFileUrl(storagePath: string) {
  const supabase = await createClient();

  const { data } = await supabase.storage
    .from("documents")
    .createSignedUrl(storagePath, 3600); // 1 hour

  return data?.signedUrl || null;
}
```

```typescript
import { inngest } from "../client";
import { createClient } from "@supabase/supabase-js";
import { resend } from "@/lib/email/client";
import { WarrantyReminderEmail } from "@/lib/email/templates/warranty-reminder";

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

export const warrantyReminders = inngest.createFunction(
  { id: "warranty-reminders", name: "Warranty Deadline Reminders" },
  { cron: "0 9 * * *" }, // Daily at 9 AM
  async ({ step }) => {
    const today = new Date();
    const in14Days = new Date(today);
    in14Days.setDate(in14Days.getDate() + 14);
    const in3Days = new Date(today);
    in3Days.setDate(in3Days.getDate() + 3);

    // Get all warranty items with upcoming or overdue deadlines
    const { data: items } = await step.run("fetch-items", async () => {
      const { data } = await supabase
        .from("home_items")
        .select(`
          id, title, registration_deadline, registration_status, home_id,
          homes(address, builder_id, handoff_status, builders(name, primary_color, contact_email)),
          home_assignments(user_id, profiles(email))
        `)
        .eq("type", "warranty")
        .eq("registration_status", "not_started")
        .not("registration_deadline", "is", null)
        .lte("registration_deadline", in14Days.toISOString().split("T")[0]);

      return data || [];
    });

    let sentCount = 0;

    for (const item of items) {
      const deadline = new Date(item.registration_deadline);
      const daysLeft = Math.ceil((deadline.getTime() - today.getTime()) / (1000 * 60 * 60 * 24));

      let reminderType: string;
      if (daysLeft < 0) reminderType = "overdue";
      else if (daysLeft <= 3) reminderType = "deadline_3d";
      else reminderType = "deadline_14d";

      const home = item.homes as any;
      const builder = home?.builders;
      const assignments = (item as any).home_assignments || [];

      for (const assignment of assignments) {
        const buyerEmail = assignment.profiles?.email;
        if (!buyerEmail) continue;

        // Check if already sent
        const { data: existing } = await supabase
          .from("reminders_sent")
          .select("id")
          .eq("home_item_id", item.id)
          .eq("reminder_type", reminderType)
          .eq("recipient_id", assignment.user_id)
          .single();

        if (existing) continue;

        await step.run(`send-${item.id}-${reminderType}-${assignment.user_id}`, async () => {
          await resend.emails.send({
            from: `${builder.name} via BuilderOS <noreply@${process.env.RESEND_DOMAIN || "resend.dev"}>`,
            to: buyerEmail,
            subject: `${daysLeft < 0 ? "Overdue: " : ""}Register your ${item.title}`,
            react: WarrantyReminderEmail({
              builderName: builder.name,
              itemTitle: item.title,
              daysLeft,
              homeAddress: home.address,
              dashboardUrl: `${process.env.NEXT_PUBLIC_APP_URL}/home/${item.home_id}`,
              primaryColor: builder.primary_color || "#1a1a1a",
            }),
          });

          await supabase.from("reminders_sent").insert({
```

```typescript
import { createClient } from "@supabase/supabase-js";

export const activationNudge = inngest.createFunction(
  { id: "activation-nudge", name: "Buyer Activation Nudge" },
  { cron: "0 10 * * *" }, // Daily at 10 AM
  async ({ step }) => {
    const threeDaysAgo = new Date();
    threeDaysAgo.setDate(threeDaysAgo.getDate() - 3);

    const { data: invitations } = await step.run("fetch-stale-invitations", async () => {
      const { data } = await supabase
        .from("invitations")
        .select("*, homes(address, builder_id, builders(name, primary_color))")
        .eq("status", "sent")

    for (const inv of invitations) {
      // Check if already nudged
      const { data: existing } = await supabase
        .from("reminders_sent")
        .select("id")
        .eq("home_id", inv.home_id)
        .eq("reminder_type", "activation_nudge")
        .eq("recipient_id", inv.email) // Using email as recipient since user may not exist yet

      const builder = home?.builders;

        const acceptUrl = `${process.env.NEXT_PUBLIC_APP_URL}/accept-invite?token=${inv.token}`;
```

```typescript
import { inngest } from "../client";
import { createClient } from "@supabase/supabase-js";
import { resend } from "@/lib/email/client";
import { BuilderEscalationEmail } from "@/lib/email/templates/builder-escalation";

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

export const builderEscalation = inngest.createFunction(
  { id: "builder-escalation", name: "Builder Escalation Digest" },
  { cron: "0 8 * * 1" }, // Weekly on Monday at 8 AM
  async ({ step }) => {
    const today = new Date();
    const todayStr = today.toISOString().split("T")[0];

    // Get all active builders
    const { data: builders } = await step.run("fetch-builders", async () => {
      const { data } = await supabase
        .from("builders")
        .select("id, name, contact_email");
      return data || [];
    });

    let sentCount = 0;

    for (const builder of builders) {
      if (!builder.contact_email) continue;

      const digest = await step.run(`digest-${builder.id}`, async () => {
        // Overdue items
        const { data: overdueItems } = await supabase
          .from("home_items")
          .select("title, registration_deadline, homes(address)")
          .eq("builder_id", builder.id)
          .eq("type", "warranty")
          .eq("registration_status", "not_started")
          .lt("registration_deadline", todayStr);

        // Stuck buyers
        const threeDaysAgo = new Date(today);
        threeDaysAgo.setDate(threeDaysAgo.getDate() - 3);

        const { data: stuckInvitations } = await supabase
          .from("invitations")
          .select("email, sent_at, homes(address)")
          .eq("builder_id", builder.id)
          .eq("status", "sent")
          .lte("sent_at", threeDaysAgo.toISOString());

        // Upcoming deadlines (next 7 days)
        const in7Days = new Date(today);
        in7Days.setDate(in7Days.getDate() + 7);

        const { data: upcomingItems } = await supabase
          .from("home_items")
          .select("title, registration_deadline, homes(address)")
          .eq("builder_id", builder.id)
          .eq("type", "warranty")
          .eq("registration_status", "not_started")
          .gte("registration_deadline", todayStr)
          .lte("registration_deadline", in7Days.toISOString().split("T")[0]);

        return {
          overdueItems: (overdueItems || []).map((i: any) => ({
            homeAddress: i.homes?.address || "",
            itemTitle: i.title,
            daysOverdue: Math.ceil((today.getTime() - new Date(i.registration_deadline).getTime()) / (1000 * 60 * 60 * 24)),
          })),
          stuckBuyers: (stuckInvitations || []).map((i: any) => ({
            email: i.email,
            homeAddress: i.homes?.address || "",
            daysSinceInvite: Math.ceil((today.getTime() - new Date(i.sent_at).getTime()) / (1000 * 60 * 60 * 24)),
          })),
          upcomingDeadlines: (upcomingItems || []).map((i: any) => ({
            homeAddress: i.homes?.address || "",
            itemTitle: i.title,
            daysLeft: Math.ceil((new Date(i.registration_deadline).getTime() - today.getTime()) / (1000 * 60 * 60 * 24)),
          })),
        };
      });

      if (digest.overdueItems.length === 0 && digest.stuckBuyers.length === 0 && digest.upcomingDeadlines.length === 0) {
        continue;
      }

      await step.run(`send-escalation-${builder.id}`, async () => {
        await resend.emails.send({
          from: `BuilderOS <noreply@${process.env.RESEND_DOMAIN || "resend.dev"}>`,
          to: builder.contact_email!,
          subject: `BuilderOS: ${digest.overdueItems.length} overdue, ${digest.stuckBuyers.length} stuck buyers`,
          react: BuilderEscalationEmail({
            builderName: builder.name,
            ...digest,
          }),
        });
      });

      sentCount++;
    }

    return { sent: sentCount };
  }
);
```

```tsx
import { createClient } from "@/lib/supabase/server";
import { getCurrentBuilder } from "@/lib/queries/builders";

import Link from "next/link";

export default async function DashboardPage() {
  const context = await getCurrentBuilder();
  if (!context) return null;

  const supabase = await createClient();
  const builderId = context.builder.id;

  // Get homes with status counts
  const { data: homes } = await supabase
    .from("homes")
    .select("id, address, lot_number, handoff_status, completion_pct, close_date, projects(name)")

  const statusCounts = allHomes.reduce((acc: Record<string, number>, h) => {

    return acc;

  const today = new Date().toISOString().split("T")[0];
  const { data: overdueItems } = await supabase
    .from("home_items")

    .eq("status", "sent")
    .limit(10);
```
